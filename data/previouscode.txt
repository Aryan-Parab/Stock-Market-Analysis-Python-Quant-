# eda.py

import pandas as pd
import numpy  as np
import matplotlib.pyplot as plt
import seaborn as sns

# Load clean dataset for eda
df = pd.read_csv('data/AAPL_cleaned.csv')

# Cleaning the dataset by dropping the 'Unnamed: 0' column
#df = df.drop(columns=['Unnamed: 0'])

# Convert date column
df['Date'] = pd.to_datetime(df['Date'])

# Set Date as index
df.set_index('Date', inplace=True)

print(df.head(3))

df.info()

# Understand price vs returns 
#Simple daily returns ( there is no fixed range for daily returns)
# if the mean daily return is 0.0005 -> 0.05% daily -> good, 0.001-> 0.1% daily -> strong, negative-> prblm

df['daily_return'] = df['Close'].pct_change()
# pct_change (returns formula) = (price today - price yesterday) / price yesterday

# log return ( very important in quants)
df['log_return'] = np.log(df['Close'] / df['Close'].shift(1))

# log_return = ln(P_today / P_yesterday) = ln(P_today) - ln(P_yesterday)
# used in quantitave finance, better statistical properities, additive over time
print(df[['Close','daily_return','log_return']].head())

# --------xxxxxxxxxxxxx------------------

# Basics Statistics
